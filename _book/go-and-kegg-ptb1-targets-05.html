type_of_GO = "BP"

assign(paste0("GO_", type_of_GO), enrichGO(
  genes,
  organismDB,
  keyType = "ENSEMBL",
  ont = type_of_GO,
  pvalueCutoff = pvalueCutoff,
  minGSSize = minGSSize,
  pAdjustMethod = "BH",
  universe = universe,
  readable = TRUE))

df_GO <- eval(parse(text = paste0("GO_", type_of_GO)))
if(nrow(df_GO) > 0){
  DT::datatable(df_GO@result %>% filter(p.adjust < 0.1), options = list(scrollX = TRUE))
} else "no GOs were found"
if(nrow(df_GO) > 0){
  barplot(df_GO, showCategory = showCategory, 
                                title = type_of_GO)
} else "no GOs to plot"
